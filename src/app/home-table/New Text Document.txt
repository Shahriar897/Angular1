  populateData(n: number){
    for(let i = 1; i<= n; i++)
      {
        let person = {
          id : i,
          name : this.randomNameGenerator(),
          fatherName : this.randomNameGenerator(),
          motherName: this.randomNameGenerator(),
          address: this.randomAddressGenerator(),
